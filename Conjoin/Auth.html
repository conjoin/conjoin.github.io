def self.setup app
  self.app = app

  if not Conjoin.env.mounted?
    require 'shield'
    app.plugin Shield::Helpers
    app.use Shield::Middleware, "/login"
  end
end

def sign_in *args
  if args.length > 1
    user, scope = args
  else
    scope = :user
    user  = args.first
  end

  if Auth.app.mounted?
    @request = req
    super scope, user
  else
    session.clear
    session['VendorWizard::User'] = user.id
  end
end
